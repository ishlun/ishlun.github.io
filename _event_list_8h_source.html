49  void Add(HookList &hooks, CallbackType const &cb, unsigned int id) {
50  bool can_add = true;
51  if (id == 0) {
52  FnPtrType const *target = cb. template target<FnPtrType>();
53  if (target)
54  can_add = !HasFunctionPtr(hooks, *target);
55  }
56  else
57  can_add = !HasFunctionId(hooks, id);
58  if (can_add) {
59  Patch();
60  hooks.emplace_back(cb, id);
61  }
62  }
63 
64  void AddBefore(CallbackType const &cb, unsigned int id) { Add(hooksBefore, cb, id); }
65  void AddAfter(CallbackType const &cb, unsigned int id) { Add(hooksAfter, cb, id); }
66 
67  void Remove(HookList &hooks, FnPtrType fn) {
68  auto end = std::remove_if(hooks.begin(), hooks.end(), [&fn](HookInfo &hook) {
69  FnPtrType *ptr = hook.first. template target<FnPtrType>();
70  return ptr && *ptr == fn;
71  });
72  hooks.erase(end, hooks.end());
73  }
74 
75  void RemoveBefore(FnPtrType fn) { Remove(hooksBefore, fn); }
76  void RemoveAfter(FnPtrType fn) { Remove(hooksAfter, fn); }
77 
78  void RemoveById(HookList &hooks, unsigned int id) {
79  if (id == 0)
80  return;
81  auto end = std::remove_if(hooks.begin(), hooks.end(), [&id](HookInfo &hook) {
82  return hook.second == id;
83  });
84  hooks.erase(end, hooks.end());
85  }
86 
87  void RemoveBeforeById(unsigned int id) { RemoveById(hooksBefore, id); }
88  void RemoveAfterById(unsigned int id) { RemoveById(hooksAfter, id); }

90 private:
91  bool bPatched = false;
92  HookList hooksBefore;
93  HookList hooksAfter;
94 
95  bool HasFunctionPtr(HookList &hooks, FnPtrType fn) {
96  for (auto &hook : hooks) {
97  FnPtrType const *ptr = hook.first. template target<FnPtrType>();
98  if (ptr && *ptr == fn)
99  return true;
100  }
101  return false;
102  }
103 
104  bool HasFunctionId(HookList &hooks, unsigned int id) {
105  for (auto &hook : hooks) {
106  if (hook.second == id)
107  return true;
108  }
109  return false;
110  }

112  void Patch() {
113  if (bPatched == false) {
114  bPatched = true;
115  PatchAll(RList());
116  }
117  }

119  void PatchAll(RefList<>) {}

121  template<int RefAddr, int GameVersion, int RefType, int RefObjectId, int RefIndexInObject, int... MoreHooks>
122  void PatchAll(RefList<RefAddr, GameVersion, RefType, RefObjectId, RefIndexInObject, MoreHooks...>) {
123  if (GameVersion == GetGameVersion()) {
124  // Using void* instead of Ret as the return value because we need to capture the return value and return that,
125  // and void* has the size of the x86/x64 (and most arches) register types, so let's use that.
126  using hook_type = Injector<std::conditional_t<RefType == H_CALL, injector::scoped_call,
127  std::conditional_t<RefType == H_JUMP, injector::scoped_jmp, injector::scoped_callback>>,
128  RefAddr, void*(Args...)>;
129  injector::make_static_hook<hook_type>([this](typename hook_type::func_type func, Args... args) {
130  auto arg_tie = std::forward_as_tuple(std::forward<Args>(args)...);
131  std::for_each(hooksBefore.begin(), hooksBefore.end(), [&](HookInfo& hook) {
132  SelectedArgPicker()(hook.first, arg_tie);
133  });
134  void* ret = func(std::forward<Args>(args)...);
135  std::for_each(hooksAfter.begin(), hooksAfter.end(), [&](HookInfo& hook) {
136  SelectedArgPicker()(hook.first, arg_tie);
137  });
138  return ret;
139  });
140  }
141  PatchAll(RefList<MoreHooks...>());
142  }
