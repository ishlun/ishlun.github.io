48  /*
49  * scoped_basic
50  * Base for scoped types which will need a buffer to save/restore stuff
51  */
52  template<size_t bufsize> // TODO specialize bufsize=0 to be dynamic
53  class scoped_basic : public scoped_base
54  {
55  private:
56  uint8_t buf[bufsize];// Saved content
57  memory_pointer_raw addr; // Data saved from this address
58  size_t size; // Size saved
59  bool saved; // Something saved?
60  bool vp; // Virtual protect?
61 
62  public:
63 
64  static const bool is_dynamic = false;
65 
66  // Restore the previosly saved data
67  // Problems may arise if someone else hooked the same place using the same method
68  virtual void restore()
69  {
70  #ifndef INJECTOR_SCOPED_NOSAVE_NORESTORE
71  if(this->saved)
72  {
73  WriteMemoryRaw(this->addr, this->buf, this->size, this->vp);
74  this->saved = false;
75  }
76  #endif
77  }
78 
79  // Save buffer at @addr with @size and virtual protect @vp
80  virtual void save(memory_pointer_tr addr, size_t size, bool vp)
81  {
82  #ifndef INJECTOR_SCOPED_NOSAVE_NORESTORE
83  assert(size <= bufsize); // Debug Safeness
84  this->restore(); // Restore anything we have saved
85  this->saved = true; // Mark that we have data save
86  this->addr = addr.get<void>(); // Save address
87  this->size = size; // Save size
88  this->vp = vp; // Save virtual protect
89  ReadMemoryRaw(addr, buf, size, vp); // Save buffer
90  #endif
91  }
92 
93  public:
94  // Constructor, initialises
95  scoped_basic() : saved(false)
96  {}
97 
98  ~scoped_basic()
99  {
100  this->restore();
101  }
102 
103  // No copy construction, we can't do this! Sure we can move construct :)
104  scoped_basic(const scoped_basic&) = delete;
105  scoped_basic(scoped_basic&& rhs)
106  {
107  *this = std::move(rhs);
108  }
109 
110  scoped_basic& operator=(const scoped_basic& rhs) = delete;
111  scoped_basic& operator=(scoped_basic&& rhs)
112  {
113  if((this->saved = rhs.saved))
114  {
115  assert(bufsize >= rhs.size);
116 
117  this->addr = rhs.addr;
118  this->size = rhs.size;
119  this->vp = rhs.vp;
120  memcpy(buf, rhs.buf, rhs.size);
121 
122  rhs.saved = false;
123  }
124  return *this;
125  }
126  };

306  /*
307  * function_hooker_manager
308  * Manages many function_hookers that points to the same address
309  * The need for this function arises because otherwise we would only be able to allow one hook per address using function_hookers
310  * This manager takes care of the amount of hooks placed in a particular address, calls the hooks and unhooks when necessary.
311  */
312  template<class CallType, class ToManage, class Ret, class ...Args>
313  class function_hooker_manager : protected CallType
314  {
315  private:
316  using func_type_raw = typename ToManage::func_type_raw;
317  using func_type = typename ToManage::func_type;
318  using functor_type = typename ToManage::functor_type;
319  using assoc_type = std::list<std::pair<const ToManage*, functor_type>>;
320 
321  // Only construction is allowed... by myself ofcourse...
322  function_hooker_manager() = default;
323  function_hooker_manager(const function_hooker_manager&) = delete;
324  function_hooker_manager(function_hooker_manager&&) = delete;
325 
326  //
327  func_type_raw original; // Pointer to the original function we've replaced
328  assoc_type assoc; // Association between owners of a hook and the hook (map)
329  bool has_hooked = false; // Is the hook already in place?
330 
331  // Find assoc iterator for the content owned by 'owned'
332  typename assoc_type::iterator find_assoc(const ToManage& owner)
333  {
334  for(auto it = assoc.begin(); it != assoc.end(); ++it)
335  if(it->first == &owner) return it;
336  return assoc.end();
337  }
338 
339  // Adds a new item to the association map (or override if already in the map)
340  void add(const ToManage& hooker, functor_type functor)
341  {
342  auto it = find_assoc(hooker);
343  if(it != assoc.end())
344  it->second = std::move(functor);
345  else
346  assoc.emplace_back(&hooker, std::move(functor));
347  }
348 
349  public:
350  // Forwards the call to all the installed hooks
351  static Ret call_hooks(Args&... args)
352  {
353  auto& manager = *instance();
354 
355  if(manager.assoc.size() == 0) // This may be uncommon but may happen (?), no hook installed
356  return manager.original(args...);
357 
358  // Functor for the original call
359  func_type original = [&manager](Args... args) -> Ret {
360  return manager.original(args...);
361  };
362 
363  if(manager.assoc.size() == 1)
364  {
365  // We have only one hook, just use it directly no need to go further in complexity
366  auto& functor = manager.assoc.begin()->second;
367  return functor(std::move(original), args...);
368  }
369  else
370  {
371  // Build a serie of functors which captures the previous functor sending it to the next functor,
372  // that's what would happen if the hooks took place independent of the template staticness (AAAAAAA)
373  func_type next = std::move(original);
374  for(auto it = manager.assoc.begin(); it != manager.assoc.end(); ++it)
375  {
376  auto& functor = it->second;
377  next = [functor, next](Args... args) -> Ret
378  {
379  return functor(next, args...);
380  };
381  }
382  return next(args...);
383  }
384  }
385 
386  public:
387 
388  // Installs a hook associated with the function_hooker 'hooker' which would call the specified 'functor'
389  // We need an auxiliar function pointer 'ptr' (to abstract calling conventions) which should forward itself to ^call_hooks
390  void install(const ToManage& hooker, functor_type functor, memory_pointer_raw ptr)
391  {
392  this->add(hooker, std::move(functor));
393 
394  // Make sure we only hook this address for the manager once
395  if(!this->has_hooked)
396  {
397  // (the following cast is needed for __thiscall functions)
398  this->original = (func_type_raw) (void*)CallType::install(plugin::GetGlobalAddress(hooker.addr), ptr).get();
399  this->has_hooked = true;
400  }
401  }
402 
403  // Restores the state of the call we've replaced in the game code
404  // All installed hooks gets uninstalled after this
405  void restore()
406  {
407  if(this->has_hooked)
408  {
409  this->has_hooked = false;
410  this->assoc.clear();
411  return CallType::restore();
412  }
413  }
414 
415  // Replaces the hook associated with 'from' to be associated with 'to'
416  // After this call the 'from' object has no association in this manager
417  void replace(const ToManage& from, const ToManage& to)
418  {
419  auto it = find_assoc(from);
420  if(it != assoc.end())
421  {
422  auto functor = std::move(it->second);
423  assoc.erase(it);
424  this->add(to, std::move(functor));
425  }
426  }
427 
428  // Removes the hook associated with the specified 'hooker'
429  // If the number of hooks reaches zero after the remotion, a restore will take place
430  void remove(const ToManage& hooker)
431  {
432  auto it = find_assoc(hooker);
433  if(it != assoc.end())
434  {
435  assoc.erase(it);
436  if(assoc.size() == 0) this->restore();
437  }
438  }
439 
440  // The instance of this specific manager
441  // This work as a shared pointer to avoid the static destruction of the manager even when a static function_hooker
442  // still wants to use it.
443  static std::shared_ptr<function_hooker_manager> instance()
444  {
445  static auto fm_ptr = std::shared_ptr<function_hooker_manager>(new function_hooker_manager());
446  return fm_ptr;
447  }
448 
449  };

452  /*
453  * function_hooker_base
454  * Base for any function_hooker, this class manages the relationship with the function hooker manager
455  */
456  template<class CallType, uintptr_t addr1, class FuncType, class Ret, class ...Args>
457  class function_hooker_base : public scoped_base
458  {
459  public:
460  static const uintptr_t addr = addr1;
461 
462  using func_type_raw = FuncType;
463  using func_type = std::function<Ret(Args...)>;
464  using functor_type = std::function<Ret(func_type, Args&...)>;
465  using manager_type = function_hooker_manager<CallType, function_hooker_base, Ret, Args...>;
466 
467  public:
468  // Constructors, move constructors, assigment operators........
469  function_hooker_base(const function_hooker_base&) = delete;
470  function_hooker_base& operator=(const function_hooker_base& rhs) = delete;
471 
472  function_hooker_base() : manager(manager_type::instance())
473  {}
474 
475  //
476  virtual ~function_hooker_base()
477  {
478  this->restore();
479  }
480 
481  // The move constructor should do a replace in the manager
482  function_hooker_base(function_hooker_base&& rhs)
483  : scoped_base(std::move(rhs)), installed(rhs.installed),
484  manager(rhs.manager) // (don't move the manager!, every function_hooker should own one)
485  {
486  manager->replace(rhs, *this);
487  }
488 
489  // The move assignment operator should also do a replace in the manager
490  function_hooker_base& operator=(function_hooker_base&& rhs)
491  {
492  scoped_base::operator=(std::move(rhs));
493  manager->replace(rhs, *this);
494  this->installed = rhs.installed;
495  this->manager = rhs.manager; // (don't move the manager! every function_hooker should own one)
496  return *this;
497  }
498 
499  // Deriveds should implement a proper install (yeah it's virtual so derived-deriveds can do some fest)
500  virtual void install(functor_type functor) = 0;
501 
502  // Restores the state of the call we've replaced in the game code
503  virtual void restore()
504  {
505  this->installed = false;
506  manager->remove(*this);
507  }
508 
509  // Checkers whether a hook is installed
510  bool has_hooked()
511  {
512  return this->installed;
513  }
514 
515  private:
516  bool installed = false; // Has a hook installed?
517  std::shared_ptr<manager_type> manager; // **EVERY** function_hooker should have a ownership over it's manager_type
518  // this prevents the static destruction of the manager_type while it may be still needed.
519 
520  protected: // Forwarders to the function hooker manager
521 
522  void install(functor_type functor, memory_pointer_raw ptr)
523  {
524  this->installed = true;
525  manager->install(*this, std::move(functor), ptr);
526  }
527 
528  static Ret call_hooks(Args&... a)
529  {
530  return manager_type::call_hooks(a...);
531  }
532  };
