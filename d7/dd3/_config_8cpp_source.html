237 void config_file::prepareData() {
238  if (_dataRead || pathEmpty())
239  return;
240  std::ifstream in;
241 #ifdef _MSC_VER
242  if (_bWidePath)
243  in.open(_widePath);
244  else
245 #endif
246  in.open(_path);
247  if (in.is_open()) {
248  unsigned int lineId = 0;
249  for (std::string line; getline(in, line); ) {
250  if (line.empty()) {
251  emptyLines.push_back(lineId++);
252  continue;
253  }
254  bool usesQuotes = false;
255  std::string name;
256  std::string value;
257  std::string comment;
258  unsigned int commentOffset = 0;
259 
260  bool scanBeforeName = true;
261  bool scanName = false;
262  bool scanAfterName = false;
263  bool scanValue = false;
264  bool scanAfterValue = false;
265  bool isCommentLine = false;
266 
267  for (unsigned int i = 0; i < line.size(); i++) {
268  if (scanBeforeName) {
269  if (line[i] == '#' || line[i] == ';') {
270  isCommentLine = true;
271  break;
272  }
273  if (line[i] == '=') {
274  // incorrect line
275  continue;
276  }
277  if (line[i] != ' ' && line[i] != '\t') {
278  name.push_back(line[i]);
279  scanBeforeName = false;
280  scanName = true;
281  }
282  }
283  else if (scanName) {
284  if (line[i] == '#' || line[i] == ';') {
285  commentOffset = i;
286  comment = line.substr(i);
287  break;
288  }
289  if (line[i] == ' ' || line[i] == '\t' || line[i] == '=') {
290  scanName = false;
291  scanAfterName = true;
292  }
293  else
294  name.push_back(line[i]);
295  }
296  else if (scanAfterName) {
297  if (line[i] == '#' || line[i] == ';') {
298  commentOffset = i;
299  comment = line.substr(i);
300  break;
301  }
302  if (line[i] == '=') {
303  _useEqualitySign = true;
304  continue;
305  }
306  else if (line[i] != ' ' && line[i] != '\t') {
307  scanAfterName = false;
308  scanValue = true;
309  value.push_back(line[i]);
310  }
311  }
312  else if (scanValue) {
313  if (line[i] == '#' || line[i] == ';') {
314  commentOffset = i;
315  comment = line.substr(i);
316  break;
317  }
318  value.push_back(line[i]);
319  }
320  }
321 
322  if (isCommentLine) {
323  comments.push_back(std::make_pair(lineId, line));
324  continue;
325  }
326 
327  if (value.size() > 0) {
328  size_t l = value.find_last_not_of(" \t");
329  if (l != std::string::npos) {
330  value = value.substr(0, l + 1);
331  size_t valSize = value.size();
332  if (valSize > 1 && value[0] == '"' && value[valSize - 1] == '"')
333  usesQuotes = true;
334  }
335  else
336  value.clear();
337  }
338 
339  paramLines.emplace_back(name, value, usesQuotes);
340  config_param_line &param = paramLines.back();
341  param.comment = comment;
342  param.commentOffset = commentOffset;
343 
344  //printf("%s \"%s\"\n", param.name.c_str(), param._value.c_str());
345 
346  lineId++;
347  }
348  }
349  _dataRead = true;
350 }

352 void config_file::writeData() {
353  if (pathEmpty())
354  return;
355  std::ofstream out;
356 #ifdef _MSC_VER
357  if (_bWidePath)
358  out.open(_widePath);
359  else
360 #endif
361  out.open(_path);
362  if (out.is_open()) {
363  unsigned int maxStrLen = 0;
364  for (config_param_line &param : paramLines) {
365  if (!param.isEmpty() && !param.name.empty()) {
366  unsigned int strSz = param.name.size();
367  if (strSz > maxStrLen)
368  maxStrLen = strSz;
369  }
370  }
371  unsigned int numCommentsLeft = comments.size();
372  for (config_param_line &param : paramLines) {
373  if (!param.isEmpty() && !param.name.empty()) {
374  out << param.name;
375  unsigned int numSpaces = maxStrLen - param.name.size() + 1;
376  for (unsigned int i = 0; i < numSpaces; i++)
377  out << ' ';
378  if (_useEqualitySign)

381  out << '\n';
382  }
383  }
