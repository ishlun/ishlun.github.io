122 bool Shader::LoadFromSource(std::ifstream &file, bool bDebug) {
123  if (file.is_open()) {
124  std::vector<std::string> lines;
125  std::string line;
126  while (std::getline(file, line))
127  lines.push_back(line);
128  // close file
129  file.close();
130  // Find shader info
131  char entryPoint[128];
132  char version[16];
133 
134  // Find fx name
135  for (auto & line : lines) {
136  if (!line.compare(0, 3, "#FX")) {
137  std::string fxLine = line;
138  size_t strBegin = fxLine.find_first_of('"');
139  if (strBegin != std::string::npos) {
140  size_t strEnd = fxLine.find_first_of('"', strBegin + 1);
141  if (strEnd != std::string::npos)
142  name = fxLine.substr(strBegin + 1, strEnd - strBegin - 1);
143  }
144  break;
145  }
146  }
147  // Load vertex shader
148  for (unsigned int i = 0; i < lines.size(); i++) {
149  if (!lines[i].compare(0, 13, "#VertexShader")) {
150  if (sscanf(lines[i].c_str(), "%*s %s %s", entryPoint, version) == 2) {
151  entryPoint[strlen(entryPoint) - 1] = '\0';
152  version[strlen(version) - 1] = '\0';
153  std::string shaderCode;
154  for (unsigned int j = i + 1; j < lines.size(); j++) {
155  if (!lines[j].empty() && lines[j].at(0) == '#')
156  break;
157  else {
158  shaderCode.append(lines[j]);
159  shaderCode.append("\n");
160  }
161  }
162  char *vShaderCode = CompileShaderFromString(shaderCode.c_str(), &entryPoint[1], &version[1], bDebug);
163  if (vShaderCode) {
164  GetD3DDevice()->CreateVertexShader((DWORD *)vShaderCode, &vertexShader);
165  delete[] vShaderCode;
166  }
167  }
168  break;
169  }
170  }
171  // Load pixel shader
172  for (unsigned int i = 0; i < lines.size(); i++) {
173  if (!lines[i].compare(0, 12, "#PixelShader")) {
174  if (sscanf(lines[i].c_str(), "%*s %s %s", entryPoint, version) == 2) {
175  entryPoint[strlen(entryPoint) - 1] = '\0';
176  version[strlen(version) - 1] = '\0';
177  std::string shaderCode;
178  for (unsigned int j = i + 1; j < lines.size(); j++) {
179  if (!lines[j].empty() && lines[j].at(0) == '#')
180  break;
181  else {
182  shaderCode.append(lines[j]);
183  shaderCode.append("\n");
184  }
185  }
186  char *pShaderCode = CompileShaderFromString(shaderCode.c_str(), &entryPoint[1], &version[1], bDebug);
187  if (pShaderCode) {
188  GetD3DDevice()->CreatePixelShader((DWORD *)pShaderCode, &pixelShader);
189  delete[] pShaderCode;
190  }
191  }
192  break;
193  }
194  }
195  return true;
196  }
197  return false;
198 }

262 bool Shader::WriteToBinaryFile(std::ofstream &file) {
263  if (file.is_open()) {
264  BinaryFileHeader header;
265  strcpy(header.fxcSignature, "FXC");
266  header.fxcSignature[3] = '\0';
267  strcpy(header.pluginSdkSignature, "plugin-sdk");
268  header.pluginSdkSignature[10] = header.pluginSdkSignature[11] = '\0';
269  header.versionId = PLUGIN_SDK_SHADER_MODULE_VERSION;
270  memset(header.name, 0, 128);
271  strcpy(header.name, name.c_str());
272  header.vsOffset = header.vsSize = header.psOffset = header.psSize = 0;
273  if (vertexShader) {
274  unsigned int functionSize = 0;
275  vertexShader->GetFunction(NULL, &functionSize);
276  if (functionSize > 0) {
277  header.vsOffset = sizeof(BinaryFileHeader);
278  header.vsSize = functionSize;
279  }
280  }
281  if (pixelShader) {
282  unsigned int functionSize = 0;
283  pixelShader->GetFunction(NULL, &functionSize);
284  if (functionSize > 0) {
285  if (header.vsOffset != 0)
286  header.psOffset = header.vsOffset + header.vsSize;
287  else
288  header.psOffset = sizeof(BinaryFileHeader);
289  header.psSize = functionSize;
290  }
291  }
292  file.write(reinterpret_cast<char *>(&header), sizeof(BinaryFileHeader));
293  if (header.vsSize > 0) {
294  unsigned int functionSize = 0;
295  char *data = new char[header.vsSize];
296  vertexShader->GetFunction(data, &functionSize);
297  file.write(data, header.vsSize);
298  delete[] data;
299  }
300  if (header.psSize > 0) {
301  unsigned int functionSize = 0;
302  char *data = new char[header.psSize];
303  pixelShader->GetFunction(data, &functionSize);
304  file.write(data, header.psSize);
305  delete[] data;
306  }
307  file.close();
308  return true;
309  }
